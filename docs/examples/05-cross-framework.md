# Example 05 — Cross-Framework Interoperability

This example demonstrates the core value proposition of Borgkit: **a LangGraph agent and a Google ADK agent, built with completely different frameworks, discovering and calling each other through the mesh as if they were the same thing.**

---

## Scenario

- **ResearchAgent** (LangGraph ReAct) — searches the web, returns summaries
- **WriterAgent** (Google ADK) — takes a research summary and drafts a report
- **User** calls `draft_report("quantum computing trends")` on the mesh

Neither agent knows the other's framework. They communicate exclusively through `AgentRequest` / `AgentResponse`.

```
User
 │
 ▼
WriterAgent (Google ADK)
 │  discovers via Borgkit
 ▼
ResearchAgent (LangGraph)
 │  returns summary
 ▼
WriterAgent (Google ADK)
 │  drafts report
 ▼
User  ←  "Here is your report on quantum computing trends..."
```

---

## Step 1 — Build and wrap ResearchAgent (LangGraph)

```python
# agents/research_agent_setup.py
from langchain_core.tools import tool
from langchain_openai     import ChatOpenAI
from langgraph.prebuilt   import create_react_agent
from plugins.langgraph_plugin import wrap_langgraph

@tool
def web_search(query: str) -> str:
    """Search the web for information on a topic."""
    # In production: call a real search API
    return f"Top results for '{query}': [1] Quantum supremacy achieved... [2] New qubit designs..."

@tool
def summarise(text: str, max_words: int = 100) -> str:
    """Summarise a block of text."""
    words = text.split()[:max_words]
    return ' '.join(words) + ('...' if len(text.split()) > max_words else '')

llm   = ChatOpenAI(model="gpt-4o-mini")
graph = create_react_agent(llm, tools=[web_search, summarise])

# Wrap for Borgkit — each tool becomes one capability
research_agent = wrap_langgraph(
    graph    = graph,
    name     = "ResearchAgent",
    agent_id = "borgkit://agent/researcher",
    owner    = "0xWallet",
    tags     = ["research", "web", "langraph"],
    tools    = [web_search, summarise],   # explicit for fastest discovery
)
```

## Step 2 — Build and wrap WriterAgent (Google ADK)

```python
# agents/writer_agent_setup.py
from google.adk.agents          import Agent
from google.adk.tools           import FunctionTool
from plugins.google_adk_plugin  import wrap_google_adk

def draft_section(topic: str, research_summary: str) -> str:
    """Draft one section of a report given a topic and research summary."""
    return (
        f"## {topic.title()}\n\n"
        f"Based on recent research: {research_summary}\n\n"
        f"This represents a significant development in the field..."
    )

def format_report(title: str, sections: str) -> str:
    """Format sections into a complete report."""
    return f"# Report: {title}\n\n{sections}\n\n---\n*Generated by WriterAgent via Borgkit*"

adk_agent = Agent(
    name        = "writer_agent",
    model       = "gemini-2.0-flash",
    description = "Drafts structured reports from research summaries",
    tools       = [FunctionTool(draft_section), FunctionTool(format_report)],
)

# Wrap for Borgkit
writer_agent = wrap_google_adk(
    agent    = adk_agent,
    name     = "WriterAgent",
    agent_id = "borgkit://agent/writer",
    owner    = "0xWallet",
    tags     = ["writing", "reports", "adk"],
)
```

## Step 3 — WriterAgent calls ResearchAgent through the mesh

The WriterAgent doesn't import or reference LangGraph anywhere. It uses the Borgkit mesh:

```python
# agents/writer_agent_extended.py
import asyncio, uuid
from interfaces import AgentRequest, AgentResponse, IAgent
from discovery.local_discovery import LocalDiscovery

class WriterAgentWithDiscovery(IAgent):
    """
    WriterAgent extended to discover ResearchAgent on the mesh
    and orchestrate the full report pipeline.
    """
    agent_id = 'borgkit://agent/writer'
    owner    = '0xWallet'
    metadata = {'name': 'WriterAgent', 'version': '0.1.0', 'tags': ['writing', 'adk']}

    def __init__(self, wrapped_adk_agent):
        self._adk = wrapped_adk_agent   # the wrapped Google ADK agent

    def get_capabilities(self):
        return ['draft_report']

    async def handle_request(self, req: AgentRequest) -> AgentResponse:
        if req.capability == 'draft_report':
            topic  = req.payload.get('topic', 'general')
            report = await self._pipeline(topic)
            return AgentResponse.success(req.request_id, {'report': report})
        return AgentResponse.error(req.request_id, f'Unknown: {req.capability}')

    async def _pipeline(self, topic: str) -> str:
        registry = LocalDiscovery.get_instance()

        # ── Step 1: find a research agent on the mesh ─────────────────────────
        researchers = await registry.query('web_search')
        if not researchers:
            return f'No research agents available — cannot draft report on {topic}'

        peer = researchers[0]
        print(f'[WriterAgent] Using researcher: {peer.name} (framework-agnostic!)')

        # ── Step 2: call it with a standard AgentRequest ──────────────────────
        #    WriterAgent has no idea this is a LangGraph agent underneath
        search_req = AgentRequest(
            request_id = str(uuid.uuid4()),
            from_id    = self.agent_id,
            capability = 'web_search',
            payload    = {'query': f'{topic} latest developments 2026'},
        )

        # In production: HTTP call to peer.network.host:peer.network.port
        # In this example: direct in-process call via stored reference
        researcher_obj = _live_agents[peer.agent_id]
        search_resp    = await researcher_obj.handle_request(search_req)

        if search_resp.status == 'error':
            return f'Research failed: {search_resp.error_message}'

        summary = search_resp.result.get('content', '')

        # ── Step 3: use ADK agent to write the report ─────────────────────────
        write_req = AgentRequest(
            request_id = str(uuid.uuid4()),
            from_id    = self.agent_id,
            capability = 'draft_section',
            payload    = {'topic': topic, 'research_summary': summary},
        )
        write_resp = await self._adk.handle_request(write_req)

        return write_resp.result.get('content', 'Failed to draft report')
```

## Step 4 — Run the full cross-framework pipeline

```python
# main.py
import asyncio, uuid
from discovery.local_discovery import LocalDiscovery
from interfaces.iagent_discovery import DiscoveryEntry, NetworkInfo, HealthStatus
from interfaces import AgentRequest
from datetime import datetime, timezone

from agents.research_agent_setup  import research_agent
from agents.writer_agent_extended import WriterAgentWithDiscovery, _live_agents
from agents.writer_agent_setup    import writer_agent as wrapped_adk

async def register(agent, port: int):
    registry = LocalDiscovery.get_instance()
    await registry.register(DiscoveryEntry(
        agent_id     = agent.agent_id,
        name         = agent.metadata.get('name', ''),
        owner        = agent.owner,
        capabilities = agent.get_capabilities(),
        network      = NetworkInfo(protocol='http', host='localhost', port=port),
        health       = HealthStatus(status='healthy',
                                     last_heartbeat=datetime.now(timezone.utc).isoformat()),
        registered_at = datetime.now(timezone.utc).isoformat(),
    ))
    _live_agents[agent.agent_id] = agent
    print(f'[Borgkit] {agent.metadata.get("name")} registered  caps={agent.get_capabilities()}')

async def main():
    # Register both framework-wrapped agents
    await register(research_agent, port=8081)   # LangGraph
    await register(wrapped_adk,    port=8082)   # Google ADK

    # Create the orchestrating WriterAgent
    writer = WriterAgentWithDiscovery(wrapped_adk)
    await register(writer, port=8080)

    # User calls draft_report — triggers LangGraph → ADK pipeline internally
    req = AgentRequest(
        request_id = str(uuid.uuid4()),
        from_id    = '0xUser',
        capability = 'draft_report',
        payload    = {'topic': 'quantum computing trends'},
    )
    resp = await writer.handle_request(req)
    print('\n=== Generated Report ===')
    print(resp.result.get('report'))

asyncio.run(main())
```

## Key takeaways

| What you get | How |
|---|---|
| **Framework isolation** | Each agent runs its native framework internally; only `AgentRequest` / `AgentResponse` cross boundaries |
| **Dynamic discovery** | WriterAgent finds ResearchAgent at runtime via capability query — no hardcoded URLs |
| **Zero framework coupling** | WriterAgent (ADK) calls ResearchAgent (LangGraph) without importing LangGraph |
| **Swappable implementations** | Replace LangGraph with CrewAI or AutoGen behind the same capability name — callers are unaffected |
| **Verifiable identity** | Both agents publish ANR records; signatures can be verified before trusting a response |
