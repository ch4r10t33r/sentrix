# x402 Payments Add-on

The `x402` add-on enables agents to charge other agents for their capabilities using the [x402 micropayment protocol](https://x402.org).

Callers that don't include a valid payment receive an HTTP-402-style `payment_required` response. The x402 client handles payment negotiation automatically.

---

## How it works

```
Caller                          Server Agent
  │                                  │
  │──── AgentRequest ───────────────>│
  │                                  │ (no payment proof)
  │<─── payment_required ────────────│
  │     + X402PaymentRequirements    │
  │                                  │
  │  [sign EIP-3009 authorisation]   │
  │                                  │
  │──── AgentRequest ───────────────>│
  │     + X402Payment proof          │
  │                                  │ [verify proof]
  │<─── AgentResponse (success) ─────│
```

## Server side — charging for capabilities

### Python

```python
from addons.x402        import X402ServerMixin, CapabilityPricing
from interfaces         import IAgent, AgentRequest, AgentResponse

class ImageGeneratorAgent(X402ServerMixin, IAgent):
    agent_id = "borgkit://agent/image-gen"
    owner    = "0xMyWallet"

    # ── pricing table ──────────────────────────────────────────────────
    x402_pricing = {
        "generate_image": CapabilityPricing.usdc_base(
            amount_usd_cents=50,        # $0.50 USDC on Base
            pay_to="0xMyWallet",
            description="AI image generation",
        ),
        "upscale_image": CapabilityPricing(
            network="base",
            asset="0x833589fcd6edb6e08f4c7c32d4f71b54bda02913",  # USDC
            amount="200000",            # $0.20
            pay_to="0xMyWallet",
        ),
        # Capabilities not in x402_pricing are served free of charge
        "get_styles": None,             # free
    }

    def get_capabilities(self):
        return ["generate_image", "upscale_image", "get_styles"]

    # ── your actual logic goes here ────────────────────────────────────
    async def _handle_paid_request(self, req: AgentRequest) -> AgentResponse:
        if req.capability == "generate_image":
            url = await _run_diffusion(req.payload["prompt"])
            return AgentResponse.success(req.request_id, {"url": url})
        if req.capability == "upscale_image":
            url = await _upscale(req.payload["url"])
            return AgentResponse.success(req.request_id, {"url": url})
        if req.capability == "get_styles":
            return AgentResponse.success(req.request_id, {"styles": ["photorealistic", "anime"]})
        return AgentResponse.error(req.request_id, "Unknown capability")
```

Or use the class-based `X402Agent`:

```typescript
import { X402Agent, usdcBase } from './addons/x402';

class ImageGeneratorAgent extends X402Agent {
  readonly agentId = 'borgkit://agent/image-gen';
  readonly owner   = '0xMyWallet';

  protected pricing = {
    generate_image: usdcBase(50, '0xMyWallet'),
  };

  getCapabilities() { return ['generate_image']; }

  protected async handlePaidRequest(req: AgentRequest): Promise<AgentResponse> {
    const url = await runDiffusion(req.payload.prompt as string);
    return { requestId: req.requestId, status: 'success', result: { url } };
  }
}
```

### Python (production wallet)

```python
from addons.x402        import X402Client
from addons.x402.client import WalletProvider, X402Payment, X402PaymentRequirements

class CoinbaseWallet(WalletProvider):
    def __init__(self, private_key_hex: str):
        self._key = private_key_hex

    def address(self) -> str:
        from identity.provider import RawKeyIdentity
        return RawKeyIdentity(self._key).owner()

    async def sign_payment(
        self, requirements: X402PaymentRequirements, original_req
    ) -> X402Payment:
        # Sign EIP-3009 transferWithAuthorization
        # See: https://eips.ethereum.org/EIPS/eip-3009
        payload   = await _sign_eip3009(requirements, self._key)
        signature = await _sign_outer(payload, self._key)
        return X402Payment(
            x402_version=1,
            scheme=requirements.scheme,
            network=requirements.network,
            payload=payload,
            signature=signature,
        )

wallet = CoinbaseWallet(private_key_hex=os.environ["CALLER_WALLET_KEY"])
client = X402Client(wallet=wallet, auto_pay=True)
resp   = await client.call(paid_agent, req)
```

## Payment verification (server side)

By default, `X402ServerMixin` runs in **dev mode** — it accepts any payment proof without verification. This is safe for development but must be replaced before going to production.

### Option 1: Use the x402 facilitator (recommended)

The facilitator is a trusted service that verifies EIP-3009 signatures without requiring you to run a blockchain node:

```python
from addons.x402 import X402Facilitator

class MyAgent(X402ServerMixin, IAgent):
    # ...
    async def _verify_x402_payment(self, payment, pricing, req):
        fac = X402Facilitator(base_url="https://x402.org/facilitator")
        return await fac.verify(payment, pricing.to_requirements(memo=req.request_id))
```

```typescript
import { withX402Payment } from './addons/x402';
import { X402Facilitator } from './addons/x402/facilitator';

const fac   = new X402Facilitator();
const agent = withX402Payment(new MyAgent(), {
  pricing: { ... },
  verify:  (payment, req) => fac.verify(payment, req),
});
```

### Option 2: On-chain verification


| Base Sepolia | 84532 | `0x036CbD53842c5426634e7929541eC2318f3dCF7e` |

| Polygon | 137 | `0x3c499c542cEF5E3811e1192ce70d8cC03d5c3359` |

Any ERC-20 token and ETH (native) are supported. USDC on Base is the recommended default for low fees and stable pricing.

## AgentRequest / AgentResponse changes

The x402 add-on uses two additional fields (both optional, so existing code is unaffected):

### AgentRequest

```python
@dataclass
class AgentRequest:
    ...
    x402: Optional[X402Payment] = None   # payment proof (client → server)
```

### AgentResponse

```python
@dataclass
class AgentResponse:
    ...
    # status can now also be "payment_required"
    status: str  # "success" | "error" | "payment_required"
    payment_requirements: Optional[List[dict]] = None  # only when status="payment_required"
```

## Full example

```python
# agents/paid_search_agent.py
import asyncio, uuid
from addons.x402          import X402ServerMixin, CapabilityPricing
from interfaces           import IAgent, AgentRequest, AgentResponse
from discovery.local_discovery import LocalDiscovery

class PaidSearchAgent(X402ServerMixin, IAgent):
    agent_id = "borgkit://agent/paid-search"
    owner    = "0xMyWallet"

    x402_pricing = {
        "deep_search": CapabilityPricing.usdc_base(10, "0xMyWallet", "Deep search — $0.10"),
    }

    def get_capabilities(self):
        return ["deep_search", "quick_search"]   # quick_search is free

    async def _handle_paid_request(self, req):
        if req.capability == "deep_search":
            results = await _run_deep_search(req.payload["query"])
            return AgentResponse.success(req.request_id, {"results": results})
        if req.capability == "quick_search":
            return AgentResponse.success(req.request_id, {"results": ["quick result"]})
        return AgentResponse.error(req.request_id, "Unknown")


# --- caller ---
async def main():
    from addons.x402        import X402Client
    from addons.x402.client import MockWalletProvider

    agent  = PaidSearchAgent()
    client = X402Client(wallet=MockWalletProvider(), auto_pay=True)

    req  = AgentRequest(
        request_id=str(uuid.uuid4()),
        from_id="borgkit://agent/caller",
        capability="deep_search",
        payload={"query": "quantum computing"},
    )
    resp = await client.call(agent, req)
    print(resp.result)

asyncio.run(main())
```
